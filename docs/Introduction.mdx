# Introduction

**React Downscreen ⛹️** - Simple framework for building accessibility-friendly selects, dropdowns, autocompletes, and comboboxes in React.

## Install

> Note: React 16.8+ is required.

### With `npm`

```sh
npm i react-downscreen
```

### With `yarn`

```sh
yarn add react-downscreen
```

```jsx
import * as React from 'react';
import Downscreen, { Button, Input, Label, Menu, MenuItem } from '..';

const items = [
  {
    id: '1',
    value: 'apple',
  },
  {
    id: '2',
    value: 'banana',
  },
  {
    id: '3',
    value: 'pear',
  },
  {
    id: '4',
    value: 'watermelon',
  },
  {
    id: '5',
    value: 'lemon',
  },
  {
    id: '6',
    value: 'orange',
  },
  {
    id: '7',
    value: 'dragonfruit',
  },
  {
    id: '8',
    value: 'papaya',
  },
];

<Downscreen
  onSelect={console.log}
  itemsLength={items.length}
  id="quick-button-example"
>
  <Button>
    {selectedIndex =>
      selectedIndex === null ? '(nothing)' : items[selectedIndex].value
    }
  </Button>
  <Menu>
    {({ index, selectedIndex, highlightedIndex }) => (
      <MenuItem
        key={items[index].id}
        index={index}
        selected={selectedIndex === index}
        highlighted={highlightedIndex === index}
      >
        {items[index].value}
      </MenuItem>
    )}
  </Menu>
</Downscreen>

<Downscreen
  onSelect={console.log}
  itemsLength={items.length}
  id="quick-input-example"
>
  <Label>Select:</Label>
  <Input>
    {selectedIndex =>
      selectedIndex === null ? null : items[selectedIndex].value
    }
  </Input>
  <Menu>
    {({ index, selectedIndex, highlightedIndex, inputValue }) =>
      items[index].value
        .toLowerCase()
        .includes(inputValue.toLowerCase()) && (
        <MenuItem
          key={items[index].id}
          index={index}
          selected={selectedIndex === index}
          highlighted={highlightedIndex === index}
        >
          {items[index].value}
        </MenuItem>
      )
    }
  </Menu>
</Downscreen>
```
